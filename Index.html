<script>


    // Rätsel 1

    function checkRiddle1(){

        let answer = document.getElementById("answer1").value.toLowerCase().trim();


        if(answer.includes("karte")){

            document.getElementById("result1").innerHTML =

                "<div class='success'>Richtig! 🎉</div>";


            document.getElementById("number1").classList.remove("hidden");

    let moveCount = 0;

    const movingButton = document.getElementById("movingButton");


    movingButton.addEventListener("mouseover", moveButton);

    movingButton.addEventListener("click", moveButton);


    function moveButton(){


        if(moveCount < 4){


            const x = Math.random() * 250 - 125;

            const y = Math.random() * 120 - 60;


            movingButton.style.transform =

                `translate(${x}px, ${y}px)`;


            moveCount++;


        } else {


            movingButton.innerHTML = "Okay okay 😅";


            document.getElementById("number2")

                .classList.remove("hidden");


            movingButton.disabled = true;

        }

    }
